def mag(z):
    return np.real(np.sqrt(np.conjugate(z)*z))


# Finds peak values in FFT
def crit(array):
    crit_index = []
    for i in range(1, len(array)-1):
        if (array[i]>array[i-1] and array[i]> array[i+1]):
            crit_index.append(i)
    return crit_index
